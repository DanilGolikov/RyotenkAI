# ============================================
# In-pod runner (FastAPI server)
# ============================================
# These are runtime deps for ``src/runner`` (uvicorn + FastAPI).
# Today fastapi + uvicorn arrive transitively via ``mlflow-skinny``,
# but that's incidental — pinning them explicitly removes the
# "works only because of mlflow's deps" coincidence and means the
# runner won't silently break when mlflow drops the FastAPI dep.
#
# python-multipart is REQUIRED by ``src/runner/api/jobs.py`` for
# the multipart ``POST /jobs`` endpoint (plugins ZIP upload).
# FastAPI doesn't pull it in by default and will RuntimeError at
# decorator-time if missing — that's the failure that killed
# every recent training run before this pin was added.
fastapi>=0.115.0,<1.0.0
uvicorn>=0.30.0,<1.0.0
python-multipart>=0.0.20,<1.0.0
websockets>=13.0,<16.0

# ``runpod`` SDK — required even on the training pod because
# ``src/providers/runpod/training/__init__.py`` eagerly imports
# the api_client → sdk_adapter → ``import runpod`` chain at
# auto-register time. Without it the trainer's PipelineConfig
# validator rejects ``provider: runpod`` with
# CONFIG_PROVIDER_NOT_REGISTERED. A lazy-import refactor would
# avoid the dep, but the SDK is small enough (~150 KB pure Python)
# that pinning it here is the pragmatic minimum.
